## Class Templates
```c++
#include<iostream>
template<typename T>
class Triple{
public:
	//构造函数
	Triple(T val1 = T(), T val2 = T(), T val3 = T());
	void PrintAll()const;
	T MinItem()const;
private:
	T item1;
	T item2;
	T item3;
};

template<typename T>
Triple<T>::Triple(T i1, T i2, T i3){
	item1 = i1
	item2 = i2
	item3 = i3
}

template<typename T>
T Triple<T>::MinItem(){
	T minVal = item1; 
	if (item2 < minVal){
	 minVal = item2;
	 } 
	if (item3 < minVal){
	  minVal = item3; 
	 } 
	 return minVal;
}
template <typename T> 
void Triple<T>::PrintAll() const { 
	std::cout << "(" << item1 << ", " << item2 << ", " << item3 << ")" << std::endl; 
}
```
